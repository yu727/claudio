## 任务一：重写 SpectrumBars 频谱可视化

### 问题
当前 `SpectrumBars.tsx` 已经使用了 Web Audio API AnalyserNode 连接 `audioPlayer`，但效果不自然：
- attack/release 响应不够跟手
- 没有对称镜像效果
- 没有粒子/光效

### 文件
- 修改：`apps/web/src/components/SpectrumBars.tsx`

### 要求
1. **保留现有 audioPlayer 连接方式**（`audioPlayer.getAudioContext()` 和 `audioPlayer.getSourceNode()`）
2. **对称镜像频谱**：取前 32 个 bar 的数据，左右镜像显示（左侧倒序 + 右侧正序），共 64 列
3. **更快的响应**：attack=0.85（快速弹起），release=0.08（缓慢回落）
4. **HSL 渐变色彩**：低频 hue=270（紫色），高频 hue=170（青色），每个 bar 按位置插值
5. **圆角顶部** + **发光阴影**（ctx.shadowBlur）
6. **底部镜像倒影**：在频谱下方画一个 30% 高度、30% 透明度的倒影
7. **idle 模式**：当 `!active` 时，用多层正弦波生成柔和的呼吸动画
8. **峰值保持线**：在每个 bar 最高点画一条 2px 的白色半透明横线，缓慢下落

## 任务四：用户画像服务

### 新建文件
- `apps/server/src/services/profile.service.ts`

### 要求
创建一个用户画像服务，将用户偏好存储到 `~/.hermes/claudio/profile.json`：

```typescript
import { readFile, writeFile, mkdir } from "node:fs/promises";
import { homedir } from "node:os";
import { join } from "node:path";
import { existsSync } from "node:fs";

export interface UserProfile {
  // 收听统计
  listenHistory: Array<{
    songId: string;
    title: string;
    artist: string;
    ts: number;
    action: "played" | "skipped" | "liked";
  }>;
  
  // 推断的偏好
  inferredGenres: Record<string, number>;  // 流派 -> 权重（从收听历史推断）
  inferredArtists: Record<string, number>; // 艺术家 -> 权重
  
  // 情绪记录
  moodHistory: Array<{
    mood: string;  // 开心/低落/兴奋/平静/焦虑/怀念
    ts: number;
  }>;
  
  // 每日推荐
  lastRecommendDate?: string; // YYYY-MM-DD
  dailyRecommendations: Array<{
    date: string;
    songIds: string[];
  }>;
}

const DEFAULT_PROFILE: UserProfile = {
  listenHistory: [],
  inferredGenres: {},
  inferredArtists: {},
  moodHistory: [],
  dailyRecommendations: [],
};

export class ProfileService {
  private profilePath: string;
  private profile: UserProfile = { ...DEFAULT_PROFILE };
  private loaded = false;

  constructor() {
    const dir = join(homedir(), ".hermes", "claudio");
    this.profilePath = join(dir, "profile.json");
    // Ensure directory exists
    if (!existsSync(dir)) {
      import("node:fs").then(fs => fs.mkdirSync(dir, { recursive: true }));
    }
  }

  async load(): Promise<UserProfile> {
    if (this.loaded) return this.profile;
    try {
      const data = await readFile(this.profilePath, "utf-8");
      this.profile = { ...DEFAULT_PROFILE, ...JSON.parse(data) };
      this.loaded = true;
    } catch {
      this.profile = { ...DEFAULT_PROFILE };
      this.loaded = true;
    }
    return this.profile;
  }

  async save(): Promise<void> {
    try {
      await writeFile(this.profilePath, JSON.stringify(this.profile, null, 2), "utf-8");
    } catch (e) {
      console.error("[profile] save failed:", (e as Error).message);
    }
  }

  getProfile(): UserProfile {
    return this.profile;
  }

  // 记录一次收听行为
  async recordListen(songId: string, title: string, artist: string, action: "played" | "skipped" | "liked") {
    await this.load();
    this.profile.listenHistory.push({ songId, title, artist, ts: Date.now(), action });
    // 只保留最近 200 条
    if (this.profile.listenHistory.length > 200) {
      this.profile.listenHistory = this.profile.listenHistory.slice(-200);
    }
    // 更新艺术家权重
    if (action === "played" || action === "liked") {
      this.profile.inferredArtists[artist] = (this.profile.inferredArtists[artist] || 0) + (action === "liked" ? 3 : 1);
    }
    await this.save();
  }

  // 记录情绪
  async recordMood(mood: string) {
    await this.load();
    this.profile.moodHistory.push({ mood, ts: Date.now() });
    if (this.profile.moodHistory.length > 50) {
      this.profile.moodHistory = this.profile.moodHistory.slice(-50);
    }
    await this.save();
  }

  // 检查今天是否已推荐
  async hasRecommendedToday(): Promise<boolean> {
    await this.load();
    const today = new Date().toISOString().slice(0, 10);
    return this.profile.lastRecommendDate === today;
  }

  // 标记今天已推荐
  async markRecommendedToday(songIds: string[]) {
    await this.load();
    const today = new Date().toISOString().slice(0, 10);
    this.profile.lastRecommendDate = today;
    this.profile.dailyRecommendations.push({ date: today, songIds });
    // 只保留最近 30 天
    if (this.profile.dailyRecommendations.length > 30) {
      this.profile.dailyRecommendations = this.profile.dailyRecommendations.slice(-30);
    }
    await this.save();
  }

  // 获取推荐给 AI 的画像摘要
  getProfileSummary(): string {
    const p = this.profile;
    const parts: string[] = [];

    // 最常听的艺术家
    const topArtists = Object.entries(p.inferredArtists)
      .sort((a, b) => b[1] - a[1])
      .slice(0, 5)
      .map(([name, weight]) => `${name}(${weight})`);
    if (topArtists.length > 0) {
      parts.push(`用户常听艺术家：${topArtists.join("、")}`);
    }

    // 最近收听
    const recent = p.listenHistory.slice(-5);
    if (recent.length > 0) {
      parts.push(`最近收听：${recent.map(r => `${r.title}-${r.artist}`).join("、")}`);
    }

    // 最近跳过的歌
    const skips = p.listenHistory.filter(r => r.action === "skipped").slice(-3);
    if (skips.length > 0) {
      parts.push(`用户跳过的歌（不喜欢）：${skips.map(r => `${r.title}-${r.artist}`).join("、")}`);
    }

    // 最近情绪
    const recentMood = p.moodHistory.slice(-3);
    if (recentMood.length > 0) {
      parts.push(`最近情绪：${recentMood.map(m => m.mood).join("→")}`);
    }

    return parts.join("\n");
  }
}
```
